// SWAPPING of 2 numbers using 3rd variable.
#include<stdio.h>
int main()
{
int a=10,b=20,c;
printf("before swapping A=%d B=%d,a,b);
c=a;
a=b;
b=c;
printf("after swapping A=%d B=%d,a,b);
}
